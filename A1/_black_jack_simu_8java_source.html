253 public static void stand() {
254 if (simpleOutcomes())
255 return;
256
257 int playerScore = playerCards.getTotalValue();
258 if (playerScore > 21 && playerCards.getNumAces() > 0)
259 playerScore -= 10;
260
261 dealerCards.cards.set(0, dealerHiddenCard);
262
263 int dealerScore = dealerCards.getTotalValue();
264
265 while (dealerScore < 16) {
266 dealerCards.cards.add(deck.takeCard());
267 dealerScore = dealerCards.getTotalValue();
268 if (dealerScore > 21 && dealerCards.getNumAces() > 0)
269 dealerScore -= 10;
270 }
271 updateCardPanels();
272
273 if (playerScore > dealerScore) {
274 lblInfo.setText("Player wins! Profit: $" + betAmount);
275 balance += betAmount * 2;
276 lblBalanceAmount.setText(String.format("$%.2f", balance));
277 } else if (dealerScore == 21) {
278 lblInfo.setText("Dealer gets Blackjack! Loss: $" + betAmount);
279 } else if (dealerScore > 21) {
280 lblInfo.setText("Dealer goes Bust! Profit: $" + betAmount);
281 balance += betAmount * 2;
282 lblBalanceAmount.setText(String.format("$%.2f", balance));
